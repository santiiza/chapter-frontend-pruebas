declare @doc_id int,
        @au_id  int,
        @cr_id  int

select @doc_id = doc_id
  from vc_documento
 where doc_factura = '47723'

select *
  from vc_documento
 where doc_id = @doc_id

select @au_id = au_id
  from vc_autorizacion
 where doc_id = @doc_id

select @cr_id = cr_id
  from vc_credito
 where au_id = @au_id

select *
  from vc_credito
 where cr_id = @cr_id

select *
  from vc_amortizacion
 where cr_id = @cr_id
